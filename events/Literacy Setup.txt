#Scandi literacy event
country_event = {
	id = 100014
	title = "Literacy fix"
	desc = "Literacy fix for Scandi"
	fire_only_once = yes	
	trigger = {
		tag = SCA
	}
	mean_time_to_happen = {
		days = -1
	}
	option = {
		name = "Learn 2 read dude"
		any_pop = {
		    limit = {
			    OR = {
					has_pop_culture = pomeranian
					has_pop_culture = finnish
					has_pop_culture = latvian
					has_pop_culture = estonian
					has_pop_culture = scottish
				}
			}
			literacy = 0.15
		}
		any_owned = {
			any_pop = {
				limit = { 
					is_primary_culture = no
					is_accepted_culture = no
					type = bureaucrats
				}
				pop_type = artisans
			}				
		}
	}
}
